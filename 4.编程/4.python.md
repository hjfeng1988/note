# 7.装饰器，类装饰器，带参数的类装饰器
类装饰器
```
class A:
    def __init__(self, f):
        print('实例化时打印我')
        self.f = f

    def __call__(self):
        print("调用__call__()时打印我")
        return self.f()

@A
def func1():
    print("I am func1()")

print('不带参数，装饰时还没开始调用__call__()')
func1()
```
带参数的类装饰器
```
class B:
    def __init__(self, level):
        print('实例化时打印我')
        self.level = level

    def __call__(self, f):
        print('调用__call__()时打印我')
        def wrapper():
            print(f'[{self.level}]: I am wrapper()')
            return f()
        return wrapper

@B('DEBUG')
def func2():
    print("I am func2()")

print('带参数，装饰时已经开始调用__call__()')
func2()
```
